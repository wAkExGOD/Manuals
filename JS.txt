ООП:
	- 1. call:
func.call(context, arg1, arg2) - ВЫЗЫВАЕТ функцию func, где контекст вызова (this) равен context;
аргументы переданы через запятую;
аргументы можно не передавать.

	- 2. apply:
func.apply(context, [arg1, arg2]) - ВЫЗЫВАЕТ функцию func, где контекст вызова (this) равен context;
аргументы переданы в массиве;
аргументы можно не передавать.

	- 3. bind:
let f = func.bind(context, arg1, arg2) - СОЗДАЕТ экземпляр функции func, где контекст вызова (this) равен context;
аргументы переданы в массиве;
аргументы можно не передавать.
